def moving_average(values, window):
    if window <= 0:
        raise ValueError("window must be positive")
    averages = []
    for i in range(len(values) - window + 1):
        chunk = values[i:i + window]
        averages.append(sum(chunk) / window)
    return averages

data = [2, 4, 6, 8, 10, 12]
print(moving_average(data, 3))
\end{lstlisting}
